## 8-bit / 1 byte registers:
  - AL / AH
  - CL / CH
  - DL / DH
  - BL / BH
  - SPL / SPH
  - BPL / BPH
  - SIL / SIH
  - DIL / DIH
  - "Memory" Registers: R8B (...) R15B

## 16-bit / 2 byte registers:
  - AX
  - CX
  - DX
  - BX
  - SP
  - BP
  - SI
  - DI
  - "Memory" Registers: R8W (...) R15W

## 32-bit / 4 byte registers:
  - EAX
  - ECX
  - EDX
  - EBX
  - ESP
  - EBP
  - ESI
  - EDI
  - "Memory" Registers: R8D (...) R15D

## 64-bit / 8 byte registers:
  - RAX
  - RCX
  - RDX
  - RBX
  - RSP
  - RBP
  - RSI
  - RDI
  - "Memory" Registers: R8 (...) R15
